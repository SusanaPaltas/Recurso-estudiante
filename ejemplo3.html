Pull-Down

Código de referencia realizado en Arduino IDE: 

//Universidad Nacional de Loja
//Carrera de Pedagogia de la Informatica
//Realizado por: Vinicio Alexander Solano Merino
//Ciclo: Septimo 
//Fecha: 12/07/2025

const int PULSADOR = 2;// Pin donde está conectado el pulsador
const int LED = 12;// Pin donde está conectado el LED (o salida)
int estado = 0;//verificar si el boton cambia de estado : (0 o 1)

void setup() {
  pinMode(PULSADOR, INPUT);//Configura el pin para leer el estado del botón
  pinMode(LED, OUTPUT); // Configura el pin del LED como salida
 
  Serial.begin(9600);//Permite ver el estado del boton
}

void loop() {
  estado = digitalRead(PULSADOR);  // Lee el estado del botón

  Serial.println(estado);//Muestra en el Monitor en Serie el valor actual de la variable estado
  
  if (estado == HIGH) {       //Si el valor es 1 cumple las siguiente condición...
    digitalWrite(LED, HIGH);  // Enciende el LED
    delay(1000);              // Espera 1 segundo
    digitalWrite(LED, LOW);   // Apaga el LED
    delay(1000);              // Espera 1 segundo
  }
}

Simulación del juego del Dinosaurio en LCD I2C

Código realizado en Arduino IDE:

//Universidad Nacional de Loja
//Carrera de Pedagogia de la Informatica
//Realizado por: Vinicio Alexander Solano Merino
//Ciclo: Septimo 

// Librerías:
#include <LiquidCrystal_I2C.h>
#include <Wire.h>

// Dirección de la LCD I2C de 16 columnas y 2 filas
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Variables globales
int estado = 0; //Almacena el estado del boton
const int PULSADOR = 2; //Pin en donde se conecta el boton
bool estadojuego = false; //Indica si el juego esta activo
int puntaje = 0; //Puntos acumulados del jugador
bool dinosobresuelo = true; //Verifica si el dinosaurio se encuentra en la fila 2 de la LCD para colisionar....

// Caracteres personalizados del juego
byte Dinosaurio[8] = {
  0b00111, 0b00111, 0b00111, 0b10110,
  0b11110, 0b11110, 0b01110, 0b01010
};
byte DinoPieIzquierdo[8] = {
  0b00111, 0b00111, 0b00111, 0b10110,
  0b11110, 0b11110, 0b01110, 0b01000
};
byte DinoPieDerecho[8] = {
  0b00111, 0b00111, 0b00111, 0b10110,
  0b11110, 0b11110, 0b01110, 0b00010
};
byte Cactus[8] = {
  0b11011, 0b11011, 0b11011, 0b11111,
  0b01100, 0b01100, 0b01100, 0b01100
};
byte Cactus2[8] = {
  0b10101, 0b10101, 0b10101, 0b10101,
  0b11111, 0b00100, 0b00100, 0b00100
};
byte Cactus3[8] = {
  0b00110, 0b10110, 0b10110, 0b10110,
  0b11110, 0b00110, 0b00110, 0b01111
};
//Configuración inicial del juego
void setup() {
  lcd.init(); //inicializa la pantalla LCD
  lcd.backlight(); //Enciende la luz de fondo
  Serial.begin(9600); //Inicia la comunicación serial 
  pinMode(PULSADOR, INPUT); //Se configura el boton como entrada (Requiere PULL-DOWN)

  //poner un identificador a cada byte...
  lcd.createChar(0, Dinosaurio);
  lcd.createChar(1, DinoPieIzquierdo);
  lcd.createChar(2, DinoPieDerecho);
  lcd.createChar(3, Cactus);
  lcd.createChar(4, Cactus2);
  lcd.createChar(5, Cactus3);

  randomSeed(analogRead(0));
}

void loop() {
  //Si el juego no ha comenzado:
  if (!estadojuego) {
    lcd.setCursor(0, 0);
    lcd.print("Presione para");
    lcd.setCursor(0, 1);
    lcd.print("iniciar       ");
    
    //Si se presiona el botón, inicia el juego....
    if (digitalRead(PULSADOR) == HIGH) {
      estadojuego = true;
      lcd.clear();
    }
  }
  //Si el juego esta activo, se ejecuta el empezarjuego...
  if (estadojuego) {
    empezarjuego();
  }
}
//Función principal del juego
void empezarjuego() {
  const int MAX_CACTUS = 3; //NUmero de cactus
  int cactusPos[MAX_CACTUS] = {-1, -1, -1}; //Posición de cada cactus 
  int cactusTipo[MAX_CACTUS] = {0, 0, 0}; //Tipos de cactus
  int tiempoParaNuevoCactus = 0; //Tiempo para que se cree un nuevo cactus

  //Bucle
  while (true) {
    // Mostrar puntaje
    lcd.setCursor(10, 0);
    lcd.print("P:");
    lcd.setCursor(12, 0);
    lcd.print("    ");
    lcd.setCursor(12, 0);
    lcd.print(puntaje);

    // Leer botón
    estado = digitalRead(PULSADOR);

    // Mostrar dinosaurio
    if (estado == 1) {
      lcd.setCursor(0, 0);
      lcd.write(byte(0));
      lcd.setCursor(0, 1);
      lcd.print(" ");
      dinosobresuelo = false;
    } else {
      lcd.setCursor(0, 1);
      lcd.write(byte(0));
      lcd.setCursor(0, 0);
      lcd.print(" ");
      dinosobresuelo = true;
    }

    // Mover y mostrar cactus
    for (int i = 0; i < MAX_CACTUS; i++) {
      if (cactusPos[i] >= 0 && cactusPos[i] < 16) {
        lcd.setCursor(cactusPos[i], 1);
        lcd.write(byte(3 + cactusTipo[i]));

        // Colisión
        if (cactusPos[i] == 1 && dinosobresuelo) {
          lcd.clear();
          lcd.setCursor(0, 0);
          lcd.print("Haz perdido");
          lcd.setCursor(0, 1);
          lcd.print("Puntos: ");
          lcd.print(puntaje);
          delay(3000);
          lcd.clear();
          estadojuego = false;
          puntaje = 0;
          return; //Sale de la función empzarjuego()
        }
      }

      // Borrar posición anterior
      if (cactusPos[i] + 1 >= 0 && cactusPos[i] + 1 < 16) {
        lcd.setCursor(cactusPos[i] + 1, 1);
        lcd.print(" ");
      }

      cactusPos[i]--;  // mover cactus a la izquierda
    }

    // Agregar nuevo cactus con separación
    tiempoParaNuevoCactus--;
    if (tiempoParaNuevoCactus <= 0) {
      for (int i = 0; i < MAX_CACTUS; i++) {
        if (cactusPos[i] < 0) {
          cactusPos[i] = 15;
          cactusTipo[i] = random(3);
          tiempoParaNuevoCactus = random(8, 14); //Espacio aleatorio
          break;
        }
      }
    }

    puntaje++; //Se incrementa el puntaje...
    delay(150); //controla la velocidad del juego...
  }
}
